object RetrofitClient {
    private const val BASE_URL = "http://10.0.2.2:5000/"

        Retrofit.Builder()
            .baseUrl(BASE_URL)
            .addConverterFactory(GsonConverterFactory.create())
            .build()
            .create(LibraryApi::class.java)
    }
}

3. Запит для реалізації бронювання книги
Запит до ШІ: як зробити кнопку "Забронювати" у списку книг, щоб при натисканні відправлявся post-запит і список оновлювався?
Результат: ШІ пояснив, як додати параметр onReserveClick у BookItem, викликати метод reserveBook у ViewModel та оновити список після успіху. Я реалізувала це та додала обробку помилок.
Фрагмент коду, отриманий від ШІ:
fun reserveBook(bookId: Int) {
    viewModelScope.launch {
        val result = repository.createReservation(bookId)
        if (result is Result.Success) loadActiveReservations()
    }
}

5. Запит для виходу з акаунту
Запит до ШІ: як зробити кнопку "Вийти", щоб очистити збережені дані користувача і повернути на екран входу?
Результат: ШІ пояснив, як викликати DataStoreHelper.clearSession() у ViewModel та змінити стан currentScreen у MainActivity. Я реалізувала це та перевірила, що після виходу сесія очищається.
Фрагмент коду, отриманий від ШІ:
fun logout() {
    viewModelScope.launch {
        DataStoreHelper.clearSession(context)
        _session.value = null
    }
}
